        function selectOption(element) {

            const scale = element.parentElement;

            const dimension = scale.dataset.dimension;

            const question = parseInt(scale.dataset.question);

            dimData.scores[question - 1] = value;

        function calculateScores() {

            let overallScore = 0;

            let totalWeight = 0;

            

            // Calculate dimension scores

            for (const [dimension, data] of Object.entries(assessmentData)) {

                if (data.scores.length > 0) {

                    const avgScore = data.scores.reduce((a, b) => a + b, 0) / data.scores.length;

                    const scoreElement = document.getElementById(dimension + 'Score');

                    const progressElement = document.getElementById(dimension + 'Progress');

                    

                    if (scoreElement) {

                        scoreElement.textContent = avgScore.toFixed(1);

                        progressElement.style.width = (avgScore * 20) + '%';

                    }

                    

                    overallScore += avgScore * data.weight;

                    totalWeight += data.weight;

                }

            }

            

            // Calculate overall score

            if (totalWeight > 0) {

                overallScore = overallScore / totalWeight;

                document.getElementById('overallScore').textContent = overallScore.toFixed(1);

                

                // Determine maturity level

                let maturityLevel = '';

                if (overallScore < 1.5) maturityLevel = 'Initial - Ad-hoc processes';

                else if (overallScore < 2.5) maturityLevel = 'Developing - Basic awareness';

                else if (overallScore < 3.5) maturityLevel = 'Established - Defined processes';

                else if (overallScore < 4.5) maturityLevel = 'Advanced - Measured approach';

                else maturityLevel = 'Optimized - Continuous improvement';

                

                document.getElementById('maturityLevel').textContent = maturityLevel;

            }

        }
